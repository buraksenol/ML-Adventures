# Tarihler arasndan seim yapmak iin, to_datetime ile datetime referans ilemini yaparz.
data['ObservationDate'] = pd.to_datetime(data['ObservationDate'])


######################### Datetime le Confirmed Says Gruplamak: ##########################
# Elimizde olan data, ayn gnler iin ayr ayr 'Confirmed' gncellemesi yapm.
# Bu yzden ayn tarihler ierisindeki gncellenen 'Confirmed' verilerini tek bir tarihe toplayp "Tek Tarih - Toplam Veri"..
#.. olarak sunmak iin, ayn tarihlerdeki verileri toplayp tek bir gruba ykleriz.
# rnek : 
# 01-02-2020 : 20 Vaka (A ehri)
# 01-02-2020 : 17 Vaka (B ehri)
#...............................
# 01-02-202 : 37 Vaka (A + B ehirleri, o tarih ierisindeki "Toplam Vakalar")
test = data['Confirmed'].groupby(data['ObservationDate'].dt.to_period('D')).sum()
df = pd.DataFrame(data=test, columns=['Date', 'Confirmed'])
df['Date'] = df.index
df.reset_index(drop=True, inplace=True)
df['Date'] = df['Date'].dt.to_timestamp('s').dt.strftime('%m-%d-%Y')


######################## Datetime le Recovered Says Gruplamak: #########################
# Tablo zerinde daha sonra kullanrken, grsel bir anlam ifade etmesi iin, confirmed deerlerinin yannda...
# Ayn tarih ierisinde iyileen hasta says ("Recovered") deerlerinide ekleriz.
test2 = data['Recovered'].groupby(data['ObservationDate'].dt.to_period('D')).sum()
df2 = pd.DataFrame(data=test2, columns=['Date', 'Recovered'])
df2['Date'] = df2.index
df2.reset_index(drop=True, inplace=True)
df2['Date'] = df2['Date'].dt.to_timestamp('s').dt.strftime('%m-%d-%Y')



# df = Date ve Confirmed bilgilerinin bulunduu tablo.
# df2 = Date ve Recovered bilgilerinin bulunduu tablo.
# lk tablomuzdaki Confirmed zniteliinin yanna 'Recovered' zniteliinide ekleyip genel bir tablo oluturalm.
df['Recovered'] = df2['Recovered']

# Genel Tablomuz.
# Bu tabloda, Verilen tarih ierisinde, datasetimiz iinde kaydedilmi tm ehirlerin toplam...
# ..."Kantlanm Covid-19 Vakalar (Confirmed)" ve... 
# ..."Kantlanm Covid-19'ten yileenlerin (Recovered)" bilgileri bulunmaktadr.

# Google Trend API kullanarak elde ettiimiz verileri, her keyword iin ayr ayr olacak ekilde yazdralm.

for element in keywords:
    # Bu for dngs ile keyword listemizdeki elemanlar tek tek trend analizlerini yaparak tabloya yazdrrz.
    plt.plot(trend_data[element])

    # Chart zerine legend ekleyelim:
    plt.legend([element])

    # Bilgi / Altyaz

    plt.ylabel("Trend Oran")
    # Hazrlanan figure'leri .png olarak kaydetmek iin:
